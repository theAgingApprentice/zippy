336 template <class T> void printLevel(int level, bool cr, T msg, ...)
337 {
338#ifndef DISABLE_LOGGING
339 if (level > _level)
340 {
341 return;
342 }
343 if (level < LOG_LEVEL_SILENT)
344 {
345 level = LOG_LEVEL_SILENT;
346 }
347
348
349 if (_prefix != NULL)
350 {
351 _prefix(_logOutput, level);
352 }
353
354 if (_showLevel) {
355 static const char levels[] = "FEWITV";
356 _logOutput->print(levels[level - 1]);
357 _logOutput->print(": ");
358 }
359
360 va_list args;
361 va_start(args, msg);
362 print(msg, args);
363
364 if(_suffix != NULL)
365 {
366 _suffix(_logOutput, level);
367 }
368 if (cr)
369 {
370 _logOutput->print(CR);
371 }
372#endif
373 }
